Connecting to Remote Databases in R

library(keyring)
library(RPostgreSQL)

MVP

Local PostgreSQL Connections

keyring_unlock("local")
local_db <- dbConnect(
  drv = PostgreSQL(),
  user = key_get("postgresql_username", keyring = "local"),
  password = key_get("postgresql_password", keyring = "local"),
  dbname = "wrestling",
  host = "localhost"
)
keyring_lock("local")

List of Wrestlers

dbGetQuery(
  local_db,
  "SELECT * FROM wrestlers"
)

Data Types

  • id: int
  • born: chr
  • name: chr
  • gender: chr
  • win_percent: dbl
  • total_matches: int

Remote PostgreSQL Connections

keyring_unlock("remote")
PostgreSQL(max.con = 2) # Set maximum simultaneous connections to 2
remote_db <- dbConnect(
  drv = PostgreSQL(),
  user = key_get("wrestling_username", keyring = "remote"),
  password = key_get("wrestling_password", keyring = "remote"),
  dbname = "mvhbdgdk",
  host = "manny.db.elephantsql.com"
)
keyring_lock("remote")

List of Wrestlers

dbGetQuery(
  remote_db,
  "SELECT * FROM wrestlers"
)

Extensions

dbGetQuery(
  remote_db,
  "SELECT * FROM wrestlers
  WHERE born = 'United Kingdom'"
)

dbGetQuery(
  remote_db,
  "SELECT * FROM wrestlers
  WHERE born = 'Japan'
  AND total_matches >= 40"
)

dbGetQuery(
  remote_db,
  "SELECT * FROM wrestlers
  WHERE win_percent BETWEEN 50 AND 60"
)

dbGetQuery(
  remote_db,
  "SELECT * FROM wrestlers
  WHERE name LIKE 'X%'"
)

dbGetQuery(
  remote_db,
  "SELECT * FROM wrestlers
  WHERE born LIKE 'I%'"
)

dbGetQuery(
  remote_db,
  "SELECT * FROM wrestlers
  WHERE born NOT LIKE 'I%'"
)
# Disconnect from remote_db
dbDisconnect(conn = remote_db)
